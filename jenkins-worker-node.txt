
~~~~~~~~~~~~ Jenkins Controller ~~~~~~~~~~~~

~~~ Update apt ~~~
sudo apt-get update

~~~ Update Git ~~~
sudo apt-get install git -y

~~~ Install Docker Engine ~~~
sudo apt-get install ca-certificates curl gnupg lsb-release -y
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg
echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/docker-archive-keyring.gpg] https://download.docker.com/linux/ubuntu $(lsb_release -cs) stable" | sudo tee /etc/apt/sources.list.d/docker.list > /dev/null
sudo apt-get update
sudo apt-get install docker-ce docker-ce-cli containerd.io docker-compose-plugin -y

~~~ Install Jenkins Via Docker Container ~~~
docker run -d --name=Jenkins-Server --network=host -p 8080:8080 -v jenkins_home:/var/jenkins_home --restart always jenkins/jenkins:latest
sudo docker run -d --name=Jenkins-Server --network=bridge -p 9999:8080 -p 5000:5000 -v jenkins_home:/var/jenkins_home --restart always jenkins/jenkins:latest

~~~ Install AWS CLI ~~~

sudo apt install unzip
unzip awscliv2.zip
sudo ./aws/install
aws --version
